include("../Figure1d_Il12b_BSBF_WT/HiddenCode_BSBF.jl")

gene1_name = "Il12b"
gene1 = CSV.read("../../Code_Paper/CompleteSets/GeneData/Il12b_intron_IFNG.csv", DataFrames.DataFrame, missingstring = "NA")
enh1_name = "HSS1"
enh1 = CSV.read("../../Code_Paper/CompleteSets/GeneData/HSS1_IFNG.csv", DataFrames.DataFrame, missingstring = "NA")
gene2_name = "Egr2"
gene2 = CSV.read("../../Code_Paper/CompleteSets/GeneData/Egr2_intron_IFNG.csv", DataFrames.DataFrame, missingstring = "NA")
enh2_name = "Egr2 Enh"
enh2 = CSV.read("../../Code_Paper/CompleteSets/GeneData/Egr2_enh_IFNG.csv", DataFrames.DataFrame, missingstring = "NA")

g1 = make_sumaries(gene1, gene1_name, int = "r2", limit = 1)
g2 = make_sumaries(gene2, gene2_name, int = "r2", limit =1)
e11 = make_sumaries(enh1[enh1[!,:Rep].==1, :], enh1_name, int = "r2", limit = 1)
e12 = make_sumaries(enh1[enh1[!,:Rep].==2, :], enh1_name, int = "r2", limit = 1);e12[!,:Rep] = [2 for ii in 1:nrow(e12)]
e1 = vcat(e11,e12)
e2 = make_sumaries(enh2, enh2_name, int = "r2")

do_bs_tests(gene1, gene1_name)
do_bs_tests(gene2, gene2_name)
do_bs_tests(enh1, enh1_name, limits = [1,1])
do_bs_tests(enh2, enh2_name);
